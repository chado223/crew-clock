<body>
  <div class="wrap">
    <h2>Clock In / Out</h2>
    <p><b>Now:</b> {{ now.strftime("%a %b %d, %I:%M %p") }}</p>

    <label>Crew Member</label>
    <select id="crew">
      {% for name in crew %}
        <option value="{{ name }}">{{ name }}</option>
      {% endfor %}
    </select>

    <label>Job (optional)</label>
    <input id="job" placeholder="e.g., Cool Springs HOA" />

    <div>
      <button onclick="send('clock_in')">Clock In</button>
      <button onclick="send('clock_out')">Clock Out</button>
    </div>

    <p id="msg"></p>
  </div>

  <script>
    async function send(action) {
      const crew = document.getElementById('crew').value;
      const job = document.getElementById('job').value;
      const msg = document.getElementById('msg');
      msg.textContent = 'Sending...';
      msg.className = '';

      try {
        const r = await fetch('/api/clock', {
          method: 'POST',
          headers: {'Content-Type': 'application/json'},
          body: JSON.stringify({ crew_name: crew, action, job_name: job })
        });
        const j = await r.json();
        if (!j.ok) throw new Error(j.error || 'Error');
        msg.textContent = `${j.crew_name} ${j.action.replace('_', ' ')} @ ${j.time_local}` + (j.job_name ? ` (${j.job_name})` : '');
        msg.className = 'ok';
        document.getElementById('job').value = '';
      } catch (e) {
        msg.textContent = 'Failed: ' + e.message;
        msg.className = 'err';
      }
    }
  </script>
